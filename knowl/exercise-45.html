define bin_to_dec(bit_string):

---

 bit_list = bitstring as numeric list
 reverse order of bit_list
 initialize dec_int to 0
 n = length(bit_list)

---

 for i in list(0, 1,...,n - 1):

---

  dec_int = dec_int + (bit_list[i]*2^i)

---

 return dec_int
